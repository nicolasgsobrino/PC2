# Usar la imagen base de Python 3.7.7 slim
FROM python:3.7.7-slim

# Establecer el directorio de trabajo en el contenedor
WORKDIR /app

# Copiar todos los archivos de la aplicación al contenedor
COPY . /app

# Instalar las dependencias necesarias (ajustar requirements.txt si existe)
RUN pip install --no-cache-dir -r requirements.txt

# Exponer el puerto 5080 para acceder a la aplicación
EXPOSE 5080

# Definir la variable de entorno por defecto (se sobrescribirá en tiempo de ejecución)
ENV GROUP_NUM="14"

# Comando de ejecución de la aplicación, asumiendo que se ejecuta con Python
CMD ["python", "app.py"]
